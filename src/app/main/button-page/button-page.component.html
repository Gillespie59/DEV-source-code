Normal Button Github

General action buttons.

"If href has no value, then it is a button."
normal info warning danger success

To call the component 

                      
  <!-- Normal Buttons -->
  <c3m-button>normal</c3m-button>
  <c3m-button ngClass="info" >info</c3m-button>
  <c3m-button ngClass="warning" >warning</c3m-button>
  <c3m-button ngClass="danger" >danger</c3m-button>
  <c3m-button ngClass="success">success</c3m-button>

To call the component 

  import {Component, ViewEncapsulation, ElementRef} from '@angular/core';

    @Component({
      selector: 'c3m-button',
      templateUrl: './button.component.html',
      styleUrls: ['./button.component.css'],
      encapsulation: ViewEncapsulation.None
    })
    export class BtnComponent {

      constructor(private elementRef: ElementRef) {}

      /* OnClick Event */
      public OnClick() {
        /* Alert that display ClassName */
        alert('My Class is : ' + this.elementRef.nativeElement.className);
      }
  }
