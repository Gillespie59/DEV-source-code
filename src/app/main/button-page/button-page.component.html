Normal Button Github

General action buttons.

normal info warning danger success

To call the component 

                      
  <!-- Normal Buttons -->
  <c3m-btn>normal</c3m-btn>
  <c3m-btn ngClass="info" >info</c3m-btn>
  <c3m-btn ngClass="warning" >warning</c3m-btn>
  <c3m-btn ngClass="danger" >danger</c3m-btn>
  <c3m-btn ngClass="success">success</c3m-btn>

For btn.component.html 

                      
  <!-- USE "BUTTON" Tag -->
  <button>
      <!-- ng-content contains the button text -->
      <ng-content></ng-content>
  </button>

For the buttons 

                  
  /* BUTTON */
  c3m-btn button{
      color: white;
      padding: 0.5rem 1rem 0.5rem 1rem;
      border-radius: 3px;
      background: rgb(105, 105, 105);
      cursor:pointer;
      margin-bottom:1.6rem;
    }
        /* BUTTON HOVER */
        c3m-btn button:hover{
            opacity:0.8;
        }
            /* COLORS OF DIFFERENTS BUTTONS */
            c3m-btn.success button{
                background-color: #6CBE45;
            }
            c3m-btn.info button{
                background-color: #45ABBC;
            }
            c3m-btn.warning button{
                background-color: #D769A8;
            }
            c3m-btn.danger button{
                background-color: #ED9C21;
            }

For custom colors 

                  
  /* Success */
  c3m-btn.success button{
      background-color: green;
  }
  /* Info */
  c3m-btn.info button{
      background-color: #1773B2;
  }
  /* Warning */
  c3m-btn.warning button{
      background-color: orange;
  }
  /* Danger */
  c3m-btn.danger button{
      background-color: orangered;
  }

To call the component 

                      
  import {Component, ViewEncapsulation, ElementRef} from '@angular/core';

    @Component({
      selector: 'c3m-btn',
      templateUrl: './btn.component.html',
      styleUrls: ['./btn.component.css'],
      encapsulation: ViewEncapsulation.None
    })
    export class BtnComponent {

      constructor(private elementRef: ElementRef) {}

      /* OnClick Event */
      public OnClick() {
        /* Alert that display ClassName */
        alert('My Class is : ' + this.elementRef.nativeElement.className);
      }
  }

Upload Button Github

Allow the upload of multiple files.

Choose a file

To call the component 

                      
  <!-- Button Upload -->
  <c3m-btn-file>Choose a file</c3m-btn-file>

For btn-file.component.html 

                      
  <!-- Label -->
  <label for="fileToUpload"><ng-content></ng-content></label>

  <!-- USE "input" type file  -->
  <input (change)="onChange(file.files)" type="file" name="fileToUpload" #file id="fileToUpload" multiple>

  <!-- File name list -->
  <ul>
      <li *ngFor="let f of files" > { { f.name } } </li>
  </ul>

For the button

                         
  /* Label */
  c3m-btn-file label{
    color: white;
    padding: 0.5rem 1rem 0.5rem 1rem;
    border-radius: 3px;
    background: rgb(105, 105, 105);
    cursor: pointer;
    margin-bottom: 1.6rem;
  }
  /* Label Hover */
  c3m-btn-file label:hover{
      opacity:0.8;
  }

  /* Input FILE */
  c3m-btn-file input[type="file"]{
    position:absolute;
    left:-9999px;
  }

  /* Item List file name */
  c3m-btn-file ul li {
    font-family: "quicksanditalic";
    color: #676767;
    margin-top: 0.3rem;
  }
  /* First Item List file name */
  c3m-btn-file ul li:first-child {
      margin-top: 1rem;
  }

To call the component 

                      
  import { Component, ViewEncapsulation } from '@angular/core';

    @Component({
        selector: 'c3m-btn-file',
        templateUrl: './btn-file.component.html',
        styleUrls: ['./btn-file.component.css'],
        encapsulation: ViewEncapsulation.None
    })
    export class BtnFileComponent {
        /* File List */
        files: FileList;

        /* OnChange method */
        onChange(files: FileList) {
            this.files = files;
        }
  }

Submit Button Github

A standard submit input (should appear inside a form or be attached to one).

To call the component 

                      
  <!-- Button Submit -->
  <c3m-btn-submit inputValue="submit"></c3m-btn-submit>

For btn-submit.component.html 

                      
  <!-- USE "Input" type submit -->
  <input type= "submit" value="{{ inputValue } } " >

For the button

                         
  /* Input  */
  c3m-btn-submit input{
      color: white;
      padding: 0.5rem 1rem 0.5rem 1rem;
      border-radius: 3px;
      background: rgb(105, 105, 105);
      cursor: pointer;
      border: none;
      font-size: 1rem;
      font-family: "quicksandregular";
  }
    /* Input Hover */
    c3m-btn-submit input:hover{
        opacity:0.8;
    }

To call the component 

                      
  import {Component, Input, ViewEncapsulation} from '@angular/core';

      encapsulation: ViewEncapsulation.None
    })

      constructor() { }

      /* Input Submit value */

  }
